# Filter the DataFrame for the year 2011
df_2011 = df[df['Year of Birth'] == 2011]

# Group by name and sum the counts to get the total count for each name in 2011
name_total_counts_2011 = df_2011.groupby("Child's First Name")['Count'].sum()

# Sort the names by total count in descending order and get the top name
most_popular_name_2011 = name_total_counts_2011.sort_values(ascending=False).index[0]
most_popular_name_count_2011 = name_total_counts_2011.sort_values(ascending=False).iloc[0]

# Print the most popular name and its count
print(f"The most popular baby name in 2011 is '{most_popular_name_2011}' with {most_popular_name_count_2011} occurrences.")

# Optional: Visualize the top N most popular names in 2011
n = 50 # Number of top names to visualize
top_names_2011 = name_total_counts_2011.sort_values(ascending=False).head(n)

plt.figure(figsize=(12, 6)) # Increased figure size for better readability
# Updated the barplot call to use hue for coloring based on x-axis and set legend=False
sns.barplot(x=top_names_2011.index, y=top_names_2011.values, hue=top_names_2011.index, palette='viridis', legend=False)
plt.title(f'Top {n} Most Popular Baby Names in 2011 (Total Count)')
plt.xlabel('Baby Name')
plt.ylabel('Total Count')
plt.xticks(rotation=45, ha='right')
plt.tight_layout()
